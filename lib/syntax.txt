
WRITE [{AT at} | <at>] dobj
    [UNDER other_dobj]
    [NO-GAP +nogap]
    [int_format_options]
    [ext_format_options]
    [list_elements]
    [QUICKINFO info].

int_format_options [LEFT-JUSTIFIED +ljust|CENTERED +cent|RIGHT-JUSTIFIED +rjust]
    { { [EXPONENT exp]
        [NO-GROUPING +nogrp]
        [NO-SIGN +nosgn]
        [NO-ZERO +nozero]
        [CURRENCY cur]
        { { [ DECIMALS dec]
            [ROUND scale] }
          | [ UNIT unit] } }
      | { [ ENVIRONMENT TIME FORMAT +envtime]
          [ TIME ZONE tz] }
      [STYLE stl] }
    [USING { {NO EDIT MASK +nomask}|{EDIT MASK mask} }]
    [ DD/MM/YY   +fmt1| MM/DD/YY   +fmt2
    | DD/MM/YYYY +fmt3| MM/DD/YYYY +fmt4
    | DDMMYY     +fmt5| MMDDYY     +fmt6
    | YYMMDD     +fmt7].

ext_format_options
    [COLOR       {[=]{color [ON]}|OFF}|{= col}]
    [INTENSIFIED [{ON|OFF}|{= flag}]]
    [INVERSE     [{ON|OFF}|{= flag}]]
    [HOTSPOT     [{ON|OFF}|{= flag}]]
    [INPUT       [{ON|OFF}|{= flag}]]
    [FRAMES      [{ON|OFF}|{= flag}]]
    [RESET +reset].

list_elements
      {AS CHECKBOX +checkbox}
    | {AS ICON +icon}
    | {AS SYMBOL +symbol}
    | {AS LINE +line}.



REPORT rep [list_options]
           [MESSAGE-ID mid]
           [DEFINING DATABASE ldb]
           [REDUCED FUNCTIONALITY +redfun].

list_options [NO STANDARD PAGE HEADING +nohead]
    [LINE-SIZE width]
    [LINE-COUNT page_lines[(footer_lines)]].


DATA {var TYPE abap_type [LENGTH len] [DECIMALS dec]}
    [VALUE val|{IS INITIAL +initial}]
    [READ-ONLY +readonly].

DATA var { {TYPE [LINE OF +lineof] type}
           | {LIKE [LINE OF +lineof] dobj} }
           [VALUE val|{IS INITIAL +initial}]
           [READ-ONLY +readonly].

DATA ref { {TYPE REF TO type}
           | {LIKE REF TO dobj} }
           [VALUE IS INITIAL +initial]
           [READ-ONLY +readonly].

DATA BEGIN OF struc [READ-ONLY +readonly].
*     ...
*     DATA comp ...
*     INCLUDE {TYPE|STRUCTURE} ...
*     ...
DATA END OF struc.

DATA itab { {TYPE [STANDARD]|SORTED|HASHED TABLE OF [REF TO] type}
          | {LIKE [STANDARD]|SORTED|HASHED TABLE OF dobj} }
          [tabkeys]
          [INITIAL SIZE n]
          [VALUE IS INITIAL]
          [READ-ONLY].

tabkeys [WITH key].

DATA rtab {TYPE RANGE OF type}|{LIKE RANGE OF dobj}
            [INITIAL SIZE n]
            [WITH HEADER LINE]
            [VALUE IS INITIAL]
            [READ-ONLY].

*DATA dtype TYPE dbtab|view [READ-ONLY]
*                  lob_handle_type FOR lob_handle_columns
*                 [ lob_handle_type FOR lob_handle_columns
*                  ...                                   ].

DATA struc TYPE struc_type BOXED.



MOVE {[EXACT] source  TO destination}
   | {        source ?TO destination}.
